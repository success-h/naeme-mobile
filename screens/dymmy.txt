chData = async (input: string) => {
    try {
      setLoading(true);
      const response = await fetch(
        `https://naeme-api.herokuapp.com/api/events/?title=${input}`
      );
      const data = await response.json();
      console.log(data);
      if (data?.results.length <= data.count) {
        setLoading(false);
        return data?.results;
      }
    } catch (e) {
      console.log(e);
      setLoading(false);
    }
  };
